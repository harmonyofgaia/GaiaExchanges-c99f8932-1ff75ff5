  private calculateAverageAccuracy(): number {
    const activeModels = Array.from(this.aiModels.values()).filter(
      (ai) => ai.isActive,
    );

    const totalAccuracy = activeModels.reduce(
      (sum, ai) => sum + ai.accuracy,
      0,
    );
    return totalAccuracy / activeModels.length;
